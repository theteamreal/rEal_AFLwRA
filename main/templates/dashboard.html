# 1. Install dependencies (once)
#    pip install torch

import torch
import json
from main.model_factory import create_unified_model

# 2. Build the model skeleton (must match training config)
model = create_unified_model(input_dim=20, out_dim=1)

# 3. Load the downloaded checkpoint  â† replace filename as needed
with open('fedora_global_v?.json', 'r') as f:
    weights = json.load(f)

# 4. Inject weights into skeleton
state_dict = {k: torch.tensor(v) for k, v in weights.items()}
model.load_state_dict(state_dict)
model.eval()   # switch to inference mode

# 5. Predict
#    my_new_data must be a [1, 20] float32 tensor (one row, 20 features)
with torch.no_grad():
    prediction = model(my_new_data)
    print(f"Predicted value: {prediction.item():.4f}")
